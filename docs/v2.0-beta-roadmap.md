## 2. Docker 部署

### 2.1 Dockerfile

```dockerfile
FROM python:3.11-slim

WORKDIR /app

# 安装系统依赖
RUN apt-get update && apt-get install -y \
    ffmpeg \
    && rm -rf /var/lib/apt/lists/*

# 安装 Python 依赖
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 复制代码
COPY . .

# 暴露端口
EXPOSE 8000

# 启动命令
CMD ["uvicorn", "jiuge.api:app", "--host", "0.0.0.0", "--port", "8000"]
```

## 3. 命令行工具

### 3.1 命令设计

```bash
# 搜索歌曲
jiuge search "周杰伦" --platforms netease,spotify

# 播放歌曲
jiuge play netease:12345

# 控制播放
jiuge pause
jiuge resume
jiuge next
jiuge stop

# 歌单操作
jiuge playlist list
jiuge playlist create "我的歌单"
jiuge playlist add <playlist_id> <track_id>

# 同步歌单
jiuge sync netease:pl123 --to spotify

# 上传本地音乐
jiuge upload ~/Music --to netease

# 获取推荐
jiuge recommend --limit 10

# 查看统计
jiuge stats --period week

# 配置
jiuge config set netease.cookie "xxx"
jiuge config get spotify.client_id
```

### 3.2 实现

使用 `click` 或 `typer` 库：

```python
import typer

app = typer.Typer()

@app.command()
def search(query: str, platforms: str = "netease"):
    """搜索歌曲"""
    results = jiuge.search(query, platforms.split(","))
    for i, track in enumerate(results, 1):
        typer.echo(f"{i}. {track.title} - {track.artist}")

@app.command()
def play(track_id: str):
    """播放歌曲"""
    jiuge.play(track_id)
    typer.echo(f"正在播放: {track_id}")

if __name__ == "__main__":
    app()
```

## 4. 性能优化

### 4.1 缓存层

```python
import redis
import json
from functools import wraps

redis_client = redis.Redis(host='localhost', port=6379, db=0)

def cache(key_template: str, expire: int = 3600):
    """缓存装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            key = key_template.format(*args, **kwargs)
            
            # 尝试从缓存获取
            cached = redis_client.get(key)
            if cached:
                return json.loads(cached)
            
            # 执行函数
            result = await func(*args, **kwargs)
            
            # 存入缓存
            redis_client.setex(key, expire, json.dumps(result))
            
            return result
        return wrapper
    return decorator

# 使用示例
@cache("search:{0}:{1}", expire=300)
async def search(query: str, platform: str):
    # 搜索逻辑
    pass
```

### 4.2 批量请求

```python
import asyncio
from typing import List

async def batch_search(queries: List[str], platform: str):
    """批量搜索"""
    tasks = [search(q, platform) for q in queries]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    return results
```

### 4.3 并发控制

```python
from asyncio import Semaphore

class ConcurrencyLimiter:
    def __init__(self, max_concurrent: int = 10):
        self.semaphore = Semaphore(max_concurrent)
    
    async def run(self, coro):
        async with self.semaphore:
            return await coro
```

```python
from pydantic import BaseSettings, validator

class Settings(BaseSettings):
    # 平台配置
    netease_cookie: str = ""
    spotify_client_id: str = ""
    spotify_client_secret: str = ""
    
    # 缓存配置
    redis_url: str = "redis://localhost:6379"
    cache_expire: int = 3600
    
    # API 配置
    api_host: str = "0.0.0.0"
    api_port: int = 8000
    
    @validator("spotify_client_id")
    def validate_spotify(cls, v, values):
        if values.get("spotify_enabled") and not v:
            raise ValueError("Spotify client ID required when enabled")
        return v
    
    class Config:
        env_file = ".env"
        env_prefix = "JIUGE_"
```
